% https://texfaq.org/FAQ-isitanum
\def\IsPositive#1{%
  TT\fi%
  \ifcat_\ifnum0<0#1 _\else A\fi%
}
\newcommand*{\RequirePosInt}[1]{%
   \if\IsPositive{#1}%
      \typeout{Is positive integer: #1}%
   \else%
      \PackageError{requireinteger}{Positive integer expected: #1}{}%
   \fi%
}
